CREATE TABLE pessoa (
    pessoa_id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    tipo_documento VARCHAR(10) CHECK (tipo_documento IN ('CPF', 'CNPJ')),
    tipo_pessoa VARCHAR(10) CHECK (tipo_pessoa IN ('FISICA', 'JURIDICA')),
    email VARCHAR(100) UNIQUE,
    senha VARCHAR(255),
    documento VARCHAR(20),
    telefone VARCHAR(20),
    cidade VARCHAR(50),
    estado VARCHAR(50)
);

CREATE TABLE papel (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE pessoa_papel (
    pessoa_id INT,
    papel_id INT,
    PRIMARY KEY (pessoa_id, papel_id),
    FOREIGN KEY (pessoa_id) REFERENCES pessoa(pessoa_id) ON DELETE CASCADE,
    FOREIGN KEY (papel_id) REFERENCES papel(id) ON DELETE CASCADE
);

CREATE TABLE status (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE solicitacoes (
    solicitacao_id SERIAL PRIMARY KEY,
    cliente_id INT,
    vendedor_id INT,
    descricao TEXT,
    tipo VARCHAR(50),
    status_id INT,
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (cliente_id) REFERENCES pessoa(pessoa_id),
    FOREIGN KEY (vendedor_id) REFERENCES pessoa(pessoa_id),
    FOREIGN KEY (status_id) REFERENCES status(id)
);

CREATE TABLE historico_chamados (
    id SERIAL PRIMARY KEY,
    solicitacao_id INT,
    cliente_id INT,
    vendedor_id INT,
    atendente_id INT,
    descricao TEXT,
    status_id INT,
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (solicitacao_id) REFERENCES solicitacoes(solicitacao_id) ON DELETE CASCADE,
    FOREIGN KEY (cliente_id) REFERENCES pessoa(pessoa_id),
    FOREIGN KEY (vendedor_id) REFERENCES pessoa(pessoa_id),
    FOREIGN KEY (atendente_id) REFERENCES pessoa(pessoa_id),
    FOREIGN KEY (status_id) REFERENCES status(id)
);

CREATE TABLE orcamentos (
    id SERIAL PRIMARY KEY,
    solicitacao_id INT,
    valor_total NUMERIC(10,2),
    data_finalizacao DATE,
    numero_pedido VARCHAR(50),

    FOREIGN KEY (solicitacao_id) REFERENCES solicitacoes(solicitacao_id) ON DELETE CASCADE
);

CREATE TABLE historico_status (
    id SERIAL PRIMARY KEY,
    solicitacao_id INT,
    status_id INT,
    usuario_id INT,
    observacao TEXT,
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (solicitacao_id) REFERENCES solicitacoes(solicitacao_id) ON DELETE CASCADE,
    FOREIGN KEY (status_id) REFERENCES status(id),
    FOREIGN KEY (usuario_id) REFERENCES pessoa(pessoa_id)
);
